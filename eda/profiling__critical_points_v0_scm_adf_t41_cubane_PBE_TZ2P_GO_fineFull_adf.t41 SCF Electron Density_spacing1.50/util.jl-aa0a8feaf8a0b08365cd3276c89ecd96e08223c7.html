663 j, k = first(r), last(r)
664 end
665 (keepempty || n > 0) && pushfirst!(strs, SubString(str,1,n))

686 count < 0 && throw(DomainError(count, "`count` must be non-negative."))

688 e1 = nextind(str, lastindex(str)) # sizeof(str)
689 i = a = firstindex(str)
690 patterns = map(p -> _pat_replacer(first(p)), pat_repl)
691 replaces = map(last, pat_repl)
692 rs = map(patterns) do p
693 r = findnext(p, str, a)
694 if r === nothing || first(r) == 0
695 return e1+1:0
696 end
697 r isa Int && (r = r:r) # findnext / performance fix
698 return r
699 end
700 if all(>(e1), map(first, rs))
701 foreach(_free_pat_replacer, patterns)
702 return str
703 end
704 out = IOBuffer(sizehint=floor(Int, 1.2sizeof(str)))
705 while true
706 p = argmin(map(first, rs)) # TODO: or argmin(rs), to pick the shortest first match ?
707 r = rs[p]
708 j, k = first(r), last(r)
709 j > e1 && break
710 if i == a || i <= k
711 # copy out preserved portion
712 GC.@preserve str unsafe_write(out, pointer(str, i), UInt(j-i))
713 # copy out replacement string
714 _replace(out, replaces[p], str, r, patterns[p])
715 end
716 if k < j
717 i = j
718 j == e1 && break
719 k = nextind(str, j)
720 else
721 i = k = nextind(str, k)
722 end
723 n == count && break
724 let k = k
725 rs = map(patterns, rs) do p, r
726 if first(r) < k
727 r = findnext(p, str, k)
728 if r === nothing || first(r) == 0
729 return e1+1:0
730 end
731 r isa Int && (r = r:r) # findnext / performance fix
732 end
733 return r
734 end
735 end
736 n += 1
737 end

740 return String(take!(out))

744 """
745 replace(s::AbstractString, pat=>r, [pat2=>r2, ...]; [count::Integer])
746
747 Search for the given pattern `pat` in `s`, and replace each occurrence with `r`.
748 If `count` is provided, replace at most `count` occurrences.
749 `pat` may be a single character, a vector or a set of characters, a string,
750 or a regular expression.
751 If `r` is a function, each occurrence is replaced with `r(s)`
752 where `s` is the matched substring (when `pat` is a `AbstractPattern` or `AbstractString`) or
753 character (when `pat` is an `AbstractChar` or a collection of `AbstractChar`).
754 If `pat` is a regular expression and `r` is a [`SubstitutionString`](@ref), then capture group
755 references in `r` are replaced with the corresponding matched text.
756 To remove instances of `pat` from `string`, set `r` to the empty `String` (`""`).
757
758 Multiple patterns can be specified, and they will be applied left-to-right
759 simultaneously, so only one pattern will be applied to any character, and the
760 patterns will only be applied to the input text, not the replacements.
761
762 !!! compat "Julia 1.7"
763 Support for multiple patterns requires version 1.7.
764
765 # Examples
766 ```jldoctest
767 julia> replace("Python is a programming language.", "Python" => "Julia")
768 "Julia is a programming language."
769
770 julia> replace("The quick foxes run quickly.", "quick" => "slow", count=1)
771 "The slow foxes run quickly."
772
773 julia> replace("The quick foxes run quickly.", "quick" => "", count=1)
774 "The foxes run quickly."
775
776 julia> replace("The quick foxes run quickly.", r"fox(es)?" => s"bus\\1")
777 "The quick buses run quickly."
778
779 julia> replace("abcabc", "a" => "b", "b" => "c", r".+" => "a")
780 "bca"
781 ```
782 """
783 replace(s::AbstractString, pat_f::Pair...; count=typemax(Int)) =
784 replace(String(s), pat_f..., count=count)

843 hex2bytes!(dest::AbstractVector{UInt8}, itr)
844
845 Convert an iterable `itr` of bytes representing a hexadecimal string to its binary
846 representation, similar to [`hex2bytes`](@ref) except that the output is written in-place
847 to `dest`. The length of `dest` must be half the length of `itr`.
848
849 !!! compat "Julia 1.7"
850 Calling hex2bytes! with iterators producing UInt8 requires
851 version 1.7. In earlier versions, you can collect the iterable
852 before calling instead.
853 """
854 function hex2bytes!(dest::AbstractArray{UInt8}, itr)
855 isodd(length(itr)) && throw(ArgumentError("length of iterable must be even"))
856 @boundscheck 2*length(dest) != length(itr) && throw(ArgumentError("length of output array must be half of the length of input iterable"))
857 iszero(length(itr)) && return dest
858
859 next = iterate(itr)
860 @inbounds for i in eachindex(dest)
861 x,state = next::NTuple{2,Any}
862 y,state = iterate(itr, state)::NTuple{2,Any}
863 next = iterate(itr, state)
864 dest[i] = number_from_hex(x) << 4 + number_from_hex(y)
865 end
866
867 return dest
868 end

871 @inline number_from_hex(c::Char) = number_from_hex(UInt8(c))
872 @inline function number_from_hex(c::UInt8)
873 UInt8('0') <= c <= UInt8('9') && return c - UInt8('0')
874 c |= 0b0100000
875 UInt8('a') <= c <= UInt8('f') && return c - UInt8('a') + 0x0a
876 throw(ArgumentError("byte is not an ASCII hexadecimal digit"))
877 end
